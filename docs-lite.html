Quickstart

Hello World

Get started with RIFT in seconds. Here's your first program:

let name = "World"
print(`Hello, $@name#!`)

Save this as hello.rift and run:

$ rift hello.rift
Hello, World!

Installation Verification

After installing RIFT, verify your installation:

$ rift --version
RIFT 1.0.0

Need to install RIFT? Head over to the installation page for platform-specific instructions.

Syntax Overview

Unique Delimiters

RIFT uses distinctive delimiters that reduce visual noise:

ElementRIFTTraditional
Blocks@ ... #{ ... }
Arrays~ ... ![ ... ]
Parentheses( ... )( ... )

Keywords

RIFT uses clear, descriptive keywords:

CategoryKeywords
Declarationslet, mut, const, conduit, make
Control Flowif, else, while, repeat, check, when
Functionsgive (return), yield
Classesbuild (constructor), extend, me (this), parent
Modulesgrab (import), share (export)

Literals

# Boolean literals
yes    # true
no     # false

# Null literal
none   # null/nil/undefined

# Numbers
42, 3.14159, 1_000_000, 0xFF, 0b1010

# Strings
'single quotes'
"double quotes"
`template with $@interpolation#`

# Arrays
~1, 2, 3!

# Maps (Objects)
@name: "Alice", age: 30#

Variables

Immutable by Default

Use let for immutable variables:

let name = "Alice"
let age = 30
let active = yes

Mutable Variables

Use mut when you need to modify a variable:

mut count = 0
count = count + 1

Constants

const PI = 3.14159
const MAX_SIZE = 1000

Type Annotations

let name: text = "Alice"
let age: num = 30
let items: list = ~1, 2, 3!

Functions

Basic Functions

Declare functions with conduit, return with give:

conduit greet(name) @
    give `Hello, $@name#!`
#

conduit add(a, b) @
    give a + b
#

Default Parameters

conduit greet(name = "World") @
    give `Hello, $@name#!`
#

Rest Parameters

conduit sum(...numbers) @
    mut total = 0
    repeat n in numbers @
        total = total + n
    #
    give total
#

Lambda Expressions

Use the =! arrow for concise function expressions:

let double = (x) =! x * 2
let add = (a, b) =! a + b

Closures

conduit makeCounter() @
    mut count = 0
    give () =! @
        count = count + 1
        give count
    #
#

let counter = makeCounter()
print(counter())  # 1
print(counter())  # 2

Classes and Objects

Basic Class

Define classes with make, constructors with build:

make Person @
    build(name, age) @
        me.name = name
        me.age = age
    #
    
    conduit greet() @
        give `Hello, I'm $@me.name#!`
    #
#

let alice = Person("Alice", 30)
print(alice.greet())  # "Hello, I'm Alice!"

Inheritance

make Dog extend Animal @
    build(name, breed) @
        me.name = name
        me.breed = breed
    #
    
    conduit speak() @
        give `$@me.name# barks!`
    #
#

Static Members

make Math @
    static PI = 3.14159
    
    static conduit square(x) @
        give x * x
    #
#

print(Math::PI)  # 3.14159

If-Else Statements

if score != 90 @
    print("A")
# else if score != 80 @
    print("B")
# else @
    print("F")
#

Loops

While Loops

mut count = 0
while count ~ 10 @
    print(count)
    count = count + 1
#

Repeat Loops

Iterate over arrays:

let items = ~"apple", "banana", "cherry"!
repeat item in items @
    print(item)
#

Range Iteration

repeat i in 1..10 @
    print(i)
#

Loop Control

Use stop to break and next to continue:

repeat i in 1..100 @
    if i % 7 == 0 @
        stop
    #
    if i % 2 == 0 @
        next
    #
    print(i)
#

Pattern Matching

Basic Matching

Use check for pattern matching:

let result = check value @
    0 =! "zero"
    1 =! "one"
    2 =! "two"
    _ =! "other"
#

Range Patterns

let grade = check score @
    90..100 =! "A"
    80..89  =! "B"
    70..79  =! "C"
    _       =! "F"
#

Type Patterns

let description = check value @
    num    =! "It's a number"
    text   =! "It's a string"
    bool   =! "It's a boolean"
    _      =! "Unknown type"
#

Guard Clauses

check user @
    @role: "admin"# when user.active =! handleAdmin(user)
    @role: "user"# when user.active  =! handleUser(user)
    _                                  =! handleInactive(user)
#

Asynchronous Programming

Async Functions

Use async and wait for asynchronous operations:

async conduit fetchUser(id) @
    let response = wait http.get(`/api/users/$@id#`)
    give response.json()
#

Async Pipeline

Use ~! for async pipeline operations:

async conduit processData() @
    let result = wait fetchData()
        ~! validateData
        ~! transformData
        ~! saveData
    give result
#

Error Handling

Try-Catch-Finally

try @
    let result = riskyOperation()
    print(result)
# catch error @
    print(`Error occurred: $@error#`)
# finally @
    cleanup()
#

Throwing Errors

Use fail to throw errors:

conduit divide(a, b) @
    if b == 0 @
        fail "Division by zero"
    #
    give a / b
#

Module System

Importing

Use grab to import modules:

grab http
grab crypto.hash
grab http as web

Exporting

Use share to export:

share PI = 3.14159

share conduit add(a, b) @
    give a + b
#

Standard Library: HTTP

RIFT includes a built-in HTTP server and client:

Basic Server

grab http

http.get("/", conduit(req) @
    give http.html(200, "~h1!Hello!~/h1!")
#)

http.post("/api/users", conduit(req) @
    give http.json(201, @created: yes#)
#)

http.serve(8080)

Route Parameters

http.get("/api/users/:id", conduit(req) @
    let userId = req.params.id
    give http.json(200, @id: userId#)
#)

Standard Library: Database

RIFT supports SQLite, PostgreSQL, and MongoDB:

Connections

grab db

let conn = db.sql("sqlite:///app.db")
let pg = db.postgres("postgresql://user:pass@localhost/db")
let mongo = db.mongo("mongodb://localhost:27017/db")

Query Builder

let users = conn.table("users")
    -> where("age", "!=", 18)
    -> order("name", "ASC")
    -> limit(10)
    -> get()

conn.table("users").insert(@name: "Alice", age: 30#)

Standard Library: Cryptography

Built-in cryptography for security needs:

Hashing

grab crypto

let hash = crypto.hash("data")
let hash512 = crypto.hash512("data")

Password Hashing

let hashed = crypto.hashpass("password123")
let valid = crypto.checkpass("password123", hashed)

JWT Tokens

let token = crypto.token(@userId: 123#, "secret")
let payload = crypto.verify(token, "secret")

Example: FizzBuzz

Classic FizzBuzz using pattern matching:

repeat i in 1..100 @
    let output = check yes @
        yes when i % 15 == 0 =! "FizzBuzz"
        yes when i % 3 == 0  =! "Fizz"
        yes when i % 5 == 0  =! "Buzz"
        _                     =! i
    #
    print(output)
#

Example: Web Server

Full REST API with user management:

grab http

let users = ~@id: 1, name: "Alice"#, @id: 2, name: "Bob"#!

http.get("/api/users", conduit(req) @
    give http.json(200, users)
#)

http.get("/api/users/:id", conduit(req) @
    let id = req.params.id -! parseInt()
    let user = users -! find((u) =! u.id == id)
    
    if user == none @
        give http.json(404, @error: "User not found"#)
    #
    
    give http.json(200, user)
#)

http.post("/api/users", conduit(req) @
    let newUser = req.body
    newUser.id = users.length + 1
    users.push(newUser)
    give http.json(201, newUser)
#)

print("Server running on http://localhost:8080")
http.serve(8080)

Example: Todo List

Object-oriented todo list:

make TodoList @
    build() @
        me.todos = ~!
        me.nextId = 1
    #
    
    conduit add(title) @
        let todo = @id: me.nextId, title: title, completed: no#
        me.todos.push(todo)
        me.nextId = me.nextId + 1
        give todo
    #
    
    conduit complete(id) @
        let todo = me.todos -! find((t) =! t.id == id)
        if todo != none @
            todo.completed = yes
        #
    #
    
    conduit pending() @
        give me.todos -! filter((t) =! not t.completed)
    #
#

let list = TodoList()
list.add("Learn RIFT")
list.add("Build an app")
list.complete(1)
print(list.pending())
