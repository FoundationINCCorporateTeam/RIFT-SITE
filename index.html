grab json

async conduit fetchUserData(userId) @
    let response = wait http.get(`https://api.example.com/users/$@userId#`)
    give json.parse(response.body)
#

async conduit processUsers(userIds) @
    let results = ~!
    
    repeat id in userIds @
        try @
            let user = wait fetchUserData(id)
            results.push(user)
        # catch error @
            print(`Failed to fetch user $@id#: $@error#`)
        #
    #
    
    give results
#

let users = wait processUsers(~1, 2, 3, 4, 5!)
print(`Fetched $@users.length# users`)
make TodoList @
    build() @
        me.todos = ~!
        me.nextId = 1
    #
    
    conduit add(title) @
        let todo = @
            id: me.nextId,
            title: title,
            completed: no
        #
        me.todos.push(todo)
        me.nextId = me.nextId + 1
        give todo
    #
    
    conduit complete(id) @
        let todo = me.todos -! find((t) =! t.id == id)
        if todo != none @
            todo.completed = yes
        #
    #
    
    conduit list() @
        give me.todos
    #
#

let list = TodoList()
list.add("Learn RIFT")
list.add("Build an app")
list.complete(1)
# Pattern matching with guards
let grade = check score @
    90..100 =! "A"
    80..89  =! "B"
    70..79  =! "C"
    60..69  =! "D"
    _       =! "F"
#

# Type-based pattern matching
let description = check value @
    num    =! "It's a number"
    text   =! "It's a string"
    bool   =! "It's a boolean"
    list   =! "It's an array"
    map    =! "It's a map"
    _      =! "Unknown type"
#

# Destructuring with guards
check user @
    @role: "admin"# when user.active =! handleAdmin(user)
    @role: "user"# when user.active  =! handleUser(user)
    @role: "guest"#                   =! handleGuest(user)
    _                                  =! handleInactive(user)
#
